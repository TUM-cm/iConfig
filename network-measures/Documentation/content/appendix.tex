\section{Testbed Installation}

\subsection{Installation of wrk Benchmark Tool}
\begin{minted}
[
frame=lines,
framesep=2mm,
breaklines,
baselinestretch=1.2,
fontsize=\footnotesize,
]
{bash}
sudo apt-get install build-essential libssl-dev git
git clone https://github.com/wg/wrk.git
cd wrk
make
# move the executable to somewhere in your PATH
sudo cp wrk /usr/local/bin
\end{minted}

\subsection{Installation of Ubuntu Bash on Windows}
\begin{enumerate}
	\item Update and security: activate developer mode
	\item Control panel: install windows subsystem for Linux 
	(beta)
	\item Restart
	\item Search after bash and install	
\end{enumerate}
